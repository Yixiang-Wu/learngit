2. 在本地创建某文件夹tempDir
3. git init
4. git remote add origin git@github.com:Yixiang-Wu/EssentialCPP.git
5. git add * //添加所有到临时仓库
6. git commit -m "提交说明" //提交到临时仓库
7. git push -u origin master //提交到远程仓库
8. git push origin master //第一次提交要使用-u，之后不需要使用
